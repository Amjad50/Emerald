15impl DwarfUnwindRegs for UnwindRegsAarch64 {
16    fn get(&self, register: Register) -> Option<u64> {
17        match register {
18            AArch64::SP => Some(self.sp()),
19            AArch64::X29 => Some(self.fp()),
20            AArch64::X30 => Some(self.lr()),
21            _ => None,
22        }
23    }

26impl DwarfUnwinding for ArchAarch64 {
27    fn unwind_frame<F, R, UCS, ES>(
28        section: &impl UnwindSection<R>,
29        unwind_info: &UnwindTableRow<R::Offset, UCS>,
30        encoding: Encoding,
31        regs: &mut Self::UnwindRegs,
32        is_first_frame: bool,
33        read_stack: &mut F,
34    ) -> Result<UnwindResult<Self::UnwindRule>, DwarfUnwinderError>
35    where
36        F: FnMut(u64) -> Result<u64, ()>,
37        R: Reader,
38        UCS: UnwindContextStorage<R::Offset>,
39        ES: EvaluationStorage<R>,
40    {
41        let cfa_rule = unwind_info.cfa();
42        let fp_rule = unwind_info.register(AArch64::X29);
43        let lr_rule = unwind_info.register(AArch64::X30);
44
45        match translate_into_unwind_rule(cfa_rule, &fp_rule, &lr_rule) {
46            Ok(unwind_rule) => return Ok(UnwindResult::ExecRule(unwind_rule)),
47            Err(_err) => {
48                // Could not translate into a cacheable unwind rule. Fall back to the generic path.
49                // eprintln!("Unwind rule translation failed: {:?}", err);
50            }
51        }
52
53        let cfa = eval_cfa_rule::<R, _, ES>(section, cfa_rule, encoding, regs)
54            .ok_or(DwarfUnwinderError::CouldNotRecoverCfa)?;
55
56        let lr = regs.lr();
57        let fp = regs.fp();
58        let sp = regs.sp();
59
60        let (fp, lr) = if !is_first_frame {
61            if cfa <= sp {
62                return Err(DwarfUnwinderError::StackPointerMovedBackwards);
63            }
64            let fp = eval_register_rule::<R, F, _, ES>(
65                section, fp_rule, cfa, encoding, fp, regs, read_stack,
66            )
67            .ok_or(DwarfUnwinderError::CouldNotRecoverFramePointer)?;
68            let lr = eval_register_rule::<R, F, _, ES>(
69                section, lr_rule, cfa, encoding, lr, regs, read_stack,
70            )
71            .ok_or(DwarfUnwinderError::CouldNotRecoverReturnAddress)?;
72            (fp, lr)
73        } else {
74            // For the first frame, be more lenient when encountering errors.
75            // TODO: Find evidence of what this gives us. I think on macOS the prologue often has Unknown register rules
76            // and we only encounter prologues for the first frame.
77            let fp = eval_register_rule::<R, F, _, ES>(
78                section, fp_rule, cfa, encoding, fp, regs, read_stack,
79            )
80            .unwrap_or(fp);
81            let lr = eval_register_rule::<R, F, _, ES>(
82                section, lr_rule, cfa, encoding, lr, regs, read_stack,
83            )
84            .unwrap_or(lr);
85            (fp, lr)
86        };
87
88        regs.set_fp(fp);
89        regs.set_sp(cfa);
90        regs.set_lr(lr);
91
92        Ok(UnwindResult::Uncacheable(lr))
93    }
94
95    fn rule_if_uncovered_by_fde() -> Self::UnwindRule {
96        UnwindRuleAarch64::NoOpIfFirstFrameOtherwiseFp
97    }
98}

100fn register_rule_to_cfa_offset<RO: ReaderOffset>(
101    rule: &RegisterRule<RO>,
102) -> Result<Option<i64>, ConversionError> {
103    match *rule {
104        RegisterRule::Undefined | RegisterRule::SameValue => Ok(None),
105        RegisterRule::Offset(offset) => Ok(Some(offset)),
106        _ => Err(ConversionError::RegisterNotStoredRelativeToCfa),
107    }
108}

110fn translate_into_unwind_rule<RO: ReaderOffset>(
111    cfa_rule: &CfaRule<RO>,
112    fp_rule: &RegisterRule<RO>,
113    lr_rule: &RegisterRule<RO>,
114) -> Result<UnwindRuleAarch64, ConversionError> {
115    match cfa_rule {
116        CfaRule::RegisterAndOffset { register, offset } => match *register {
117            AArch64::SP => {
118                let sp_offset_by_16 =
119                    u16::try_from(offset / 16).map_err(|_| ConversionError::SpOffsetDoesNotFit)?;
120                let lr_cfa_offset = register_rule_to_cfa_offset(lr_rule)?;
121                let fp_cfa_offset = register_rule_to_cfa_offset(fp_rule)?;
122                match (lr_cfa_offset, fp_cfa_offset) {
123                    (None, Some(_)) => Err(ConversionError::RestoringFpButNotLr),
124                    (None, None) => {
125                        if let RegisterRule::Undefined = lr_rule {
126                            // If the return address is undefined, this could have two reasons:
127                            //  - The column for the return address may have been manually set to "undefined"
128                            //    using DW_CFA_undefined. This usually means that the function never returns
129                            //    and can be treated as the root of the stack.
130                            //  - The column for the return may have been omitted from the DWARF CFI table.
131                            //    Per spec (at least as of DWARF >= 3), this means that it should be treated
132                            //    as undefined. But it seems that compilers often do this when they really mean
133                            //    "same value".
134                            // Gimli follows DWARF 3 and does not differentiate between "omitted" and "undefined".
135                            Ok(
136                                UnwindRuleAarch64::OffsetSpIfFirstFrameOtherwiseStackEndsHere {
137                                    sp_offset_by_16,
138                                },
139                            )
140                        } else {
141                            Ok(UnwindRuleAarch64::OffsetSp { sp_offset_by_16 })
142                        }
143                    }
144                    (Some(lr_cfa_offset), None) => {
145                        let lr_storage_offset_from_sp_by_8 =
146                            i16::try_from((offset + lr_cfa_offset) / 8)
147                                .map_err(|_| ConversionError::LrStorageOffsetDoesNotFit)?;
148                        Ok(UnwindRuleAarch64::OffsetSpAndRestoreLr {
149                            sp_offset_by_16,
150                            lr_storage_offset_from_sp_by_8,
151                        })
152                    }
153                    (Some(lr_cfa_offset), Some(fp_cfa_offset)) => {
154                        let lr_storage_offset_from_sp_by_8 =
155                            i16::try_from((offset + lr_cfa_offset) / 8)
156                                .map_err(|_| ConversionError::LrStorageOffsetDoesNotFit)?;
157                        let fp_storage_offset_from_sp_by_8 =
158                            i16::try_from((offset + fp_cfa_offset) / 8)
159                                .map_err(|_| ConversionError::FpStorageOffsetDoesNotFit)?;
160                        Ok(UnwindRuleAarch64::OffsetSpAndRestoreFpAndLr {
161                            sp_offset_by_16,
162                            fp_storage_offset_from_sp_by_8,
163                            lr_storage_offset_from_sp_by_8,
164                        })
165                    }
166                }
167            }
168            AArch64::X29 => {
169                let lr_cfa_offset = register_rule_to_cfa_offset(lr_rule)?
170                    .ok_or(ConversionError::FramePointerRuleDoesNotRestoreLr)?;
171                let fp_cfa_offset = register_rule_to_cfa_offset(fp_rule)?
172                    .ok_or(ConversionError::FramePointerRuleDoesNotRestoreFp)?;
173                if *offset == 16 && fp_cfa_offset == -16 && lr_cfa_offset == -8 {
174                    Ok(UnwindRuleAarch64::UseFramePointer)
175                } else {
176                    let sp_offset_from_fp_by_8 = u16::try_from(offset / 8)
177                        .map_err(|_| ConversionError::SpOffsetFromFpDoesNotFit)?;
178                    let lr_storage_offset_from_fp_by_8 =
179                        i16::try_from((offset + lr_cfa_offset) / 8)
180                            .map_err(|_| ConversionError::LrStorageOffsetDoesNotFit)?;
181                    let fp_storage_offset_from_fp_by_8 =
182                        i16::try_from((offset + fp_cfa_offset) / 8)
183                            .map_err(|_| ConversionError::FpStorageOffsetDoesNotFit)?;
184                    Ok(UnwindRuleAarch64::UseFramepointerWithOffsets {
185                        sp_offset_from_fp_by_8,
186                        fp_storage_offset_from_fp_by_8,
187                        lr_storage_offset_from_fp_by_8,
188                    })
189                }
190            }
191            _ => Err(ConversionError::CfaIsOffsetFromUnknownRegister),
192        },
193        CfaRule::Expression(_) => Err(ConversionError::CfaIsExpression),
194    }
