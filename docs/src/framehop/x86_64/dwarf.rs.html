13impl DwarfUnwindRegs for UnwindRegsX86_64 {
14    fn get(&self, register: Register) -> Option<u64> {
15        match register {
16            X86_64::RA => Some(self.ip()),
17            X86_64::RSP => Some(self.sp()),
18            X86_64::RBP => Some(self.bp()),
19            _ => None,
20        }
21    }
22}

24impl DwarfUnwinding for ArchX86_64 {
25    fn unwind_frame<F, R, UCS, ES>(
26        section: &impl UnwindSection<R>,
27        unwind_info: &UnwindTableRow<R::Offset, UCS>,
28        encoding: Encoding,
29        regs: &mut Self::UnwindRegs,
30        is_first_frame: bool,
31        read_stack: &mut F,
32    ) -> Result<UnwindResult<Self::UnwindRule>, DwarfUnwinderError>
33    where
34        F: FnMut(u64) -> Result<u64, ()>,
35        R: Reader,
36        UCS: UnwindContextStorage<R::Offset>,
37        ES: EvaluationStorage<R>,
38    {
39        let cfa_rule = unwind_info.cfa();
40        let bp_rule = unwind_info.register(X86_64::RBP);
41        let ra_rule = unwind_info.register(X86_64::RA);
42
43        match translate_into_unwind_rule(cfa_rule, &bp_rule, &ra_rule) {
44            Ok(unwind_rule) => return Ok(UnwindResult::ExecRule(unwind_rule)),
45            Err(_err) => {
46                // Could not translate into a cacheable unwind rule. Fall back to the generic path.
47                // eprintln!("Unwind rule translation failed: {:?}", err);
48            }
49        }
50
51        let cfa = eval_cfa_rule::<R, _, ES>(section, cfa_rule, encoding, regs)
52            .ok_or(DwarfUnwinderError::CouldNotRecoverCfa)?;
53
54        let ip = regs.ip();
55        let bp = regs.bp();
56        let sp = regs.sp();
57
58        let new_bp = eval_register_rule::<R, F, _, ES>(
59            section, bp_rule, cfa, encoding, bp, regs, read_stack,
60        )
61        .unwrap_or(bp);
62
63        let return_address = match eval_register_rule::<R, F, _, ES>(
64            section, ra_rule, cfa, encoding, ip, regs, read_stack,
65        ) {
66            Some(ra) => ra,
67            None => {
68                read_stack(cfa - 8).map_err(|_| DwarfUnwinderError::CouldNotRecoverReturnAddress)?
69            }
70        };
71
72        if cfa == sp && return_address == ip {
73            return Err(DwarfUnwinderError::DidNotAdvance);
74        }
75        if !is_first_frame && cfa < regs.sp() {
76            return Err(DwarfUnwinderError::StackPointerMovedBackwards);
77        }
78
79        regs.set_ip(return_address);
80        regs.set_bp(new_bp);
81        regs.set_sp(cfa);
82
83        Ok(UnwindResult::Uncacheable(return_address))
84    }
85
86    fn rule_if_uncovered_by_fde() -> Self::UnwindRule {
87        UnwindRuleX86_64::JustReturnIfFirstFrameOtherwiseFp
88    }
89}

91fn register_rule_to_cfa_offset<RO: ReaderOffset>(
92    rule: &RegisterRule<RO>,
93) -> Result<Option<i64>, ConversionError> {
94    match *rule {
95        RegisterRule::Undefined | RegisterRule::SameValue => Ok(None),
96        RegisterRule::Offset(offset) => Ok(Some(offset)),
97        _ => Err(ConversionError::RegisterNotStoredRelativeToCfa),
98    }
99}

101fn translate_into_unwind_rule<RO: ReaderOffset>(
102    cfa_rule: &CfaRule<RO>,
103    bp_rule: &RegisterRule<RO>,
104    ra_rule: &RegisterRule<RO>,
105) -> Result<UnwindRuleX86_64, ConversionError> {
106    match ra_rule {
107        RegisterRule::Undefined => {
108            // No return address. This means that we've reached the end of the stack.
109            return Ok(UnwindRuleX86_64::EndOfStack);
110        }
111        RegisterRule::Offset(offset) if *offset == -8 => {
112            // This is normal case. Return address is [CFA-8].
113        }
114        RegisterRule::Offset(_) => {
115            // Unsupported, will have to use the slow path.
116            return Err(ConversionError::ReturnAddressRuleWithUnexpectedOffset);
117        }
118        _ => {
119            // Unsupported, will have to use the slow path.
120            return Err(ConversionError::ReturnAddressRuleWasWeird);
121        }
122    }
123
124    match cfa_rule {
125        CfaRule::RegisterAndOffset { register, offset } => match *register {
126            X86_64::RSP => {
127                let sp_offset_by_8 =
128                    u16::try_from(offset / 8).map_err(|_| ConversionError::SpOffsetDoesNotFit)?;
129                let fp_cfa_offset = register_rule_to_cfa_offset(bp_rule)?;
130                match fp_cfa_offset {
131                    None => Ok(UnwindRuleX86_64::OffsetSp { sp_offset_by_8 }),
132                    Some(bp_cfa_offset) => {
133                        let bp_storage_offset_from_sp_by_8 =
134                            i16::try_from((offset + bp_cfa_offset) / 8)
135                                .map_err(|_| ConversionError::FpStorageOffsetDoesNotFit)?;
136                        Ok(UnwindRuleX86_64::OffsetSpAndRestoreBp {
137                            sp_offset_by_8,
138                            bp_storage_offset_from_sp_by_8,
139                        })
140                    }
141                }
142            }
143            X86_64::RBP => {
144                let bp_cfa_offset = register_rule_to_cfa_offset(bp_rule)?
145                    .ok_or(ConversionError::FramePointerRuleDoesNotRestoreBp)?;
146                if *offset == 16 && bp_cfa_offset == -16 {
147                    Ok(UnwindRuleX86_64::UseFramePointer)
148                } else {
149                    // TODO: Maybe handle this case. This case has been observed in _ffi_call_unix64,
150                    // which has the following unwind table:
151                    //
152                    // 00000060 00000024 0000001c FDE cie=00000048 pc=000de548...000de6a6
153                    //   0xde548: CFA=reg7+8: reg16=[CFA-8]
154                    //   0xde562: CFA=reg6+32: reg6=[CFA-16], reg16=[CFA-8]
155                    //   0xde5ad: CFA=reg7+8: reg16=[CFA-8]
156                    //   0xde668: CFA=reg7+8: reg6=[CFA-16], reg16=[CFA-8]
157                    Err(ConversionError::FramePointerRuleHasStrangeBpOffset)
158                }
159            }
160            _ => Err(ConversionError::CfaIsOffsetFromUnknownRegister),
161        },
162        CfaRule::Expression(_) => Err(ConversionError::CfaIsExpression),
163    }
164}
