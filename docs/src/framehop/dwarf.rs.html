97impl<'a, R, A, UCS> DwarfUnwinder<'a, R, A, UCS>
98where
99    R: Reader,
100    A: DwarfUnwinding,
101    UCS: UnwindContextStorage<R::Offset>,

103    pub fn new(
104        unwind_section_data: R,
105        unwind_section_type: UnwindSectionType,
106        eh_frame_hdr_data: Option<&'a [u8]>,
107        unwind_context: &'a mut UnwindContext<R::Offset, UCS>,
108        bases: BaseAddresses,
109        base_svma: u64,
110    ) -> Self {
111        let eh_frame_hdr = match eh_frame_hdr_data {
112            Some(eh_frame_hdr_data) => {
113                let hdr = EhFrameHdr::new(eh_frame_hdr_data, unwind_section_data.endian());
114                match hdr.parse(&bases, 8) {
115                    Ok(hdr) => Some(hdr),
116                    Err(_) => None,
117                }
118            }
119            None => None,
120        };
121        Self {
122            unwind_section_data,
123            unwind_section_type,
124            eh_frame_hdr,
125            unwind_context,
126            bases,
127            base_svma,
128            _arch: PhantomData,
129        }
130    }

132    pub fn get_fde_offset_for_relative_address(&self, rel_lookup_address: u32) -> Option<u32> {
133        let lookup_svma = self.base_svma + rel_lookup_address as u64;
134        let eh_frame_hdr = self.eh_frame_hdr.as_ref()?;
135        let table = eh_frame_hdr.table()?;
136        let fde_ptr = table.lookup(lookup_svma, &self.bases).ok()?;
137        let fde_offset = table.pointer_to_offset(fde_ptr).ok()?;
138        fde_offset.0.into_u64().try_into().ok()
139    }

141    pub fn unwind_frame_with_fde<F, ES>(
142        &mut self,
143        regs: &mut A::UnwindRegs,
144        is_first_frame: bool,
145        rel_lookup_address: u32,
146        fde_offset: u32,
147        read_stack: &mut F,
148    ) -> Result<UnwindResult<A::UnwindRule>, DwarfUnwinderError>
149    where
150        F: FnMut(u64) -> Result<u64, ()>,
151        ES: EvaluationStorage<R>,
152    {
153        let lookup_svma = self.base_svma + rel_lookup_address as u64;
154        let unwind_section_data = self.unwind_section_data.clone();
155        match self.unwind_section_type {
156            UnwindSectionType::EhFrame => {
157                let mut eh_frame = EhFrame::from(unwind_section_data);
158                eh_frame.set_address_size(8);
159                let unwind_info = self.unwind_info_for_fde(&eh_frame, lookup_svma, fde_offset);
160                if let Err(DwarfUnwinderError::UnwindInfoForAddressFailed(_)) = unwind_info {
161                    return Ok(UnwindResult::ExecRule(A::rule_if_uncovered_by_fde()));
162                }
163                let (unwind_info, encoding) = unwind_info?;
164                A::unwind_frame::<F, R, UCS, ES>(
165                    &eh_frame,
166                    unwind_info,
167                    encoding,
168                    regs,
169                    is_first_frame,
170                    read_stack,
171                )
172            }
173            UnwindSectionType::DebugFrame => {
174                let mut debug_frame = DebugFrame::from(unwind_section_data);
175                debug_frame.set_address_size(8);
176                let unwind_info = self.unwind_info_for_fde(&debug_frame, lookup_svma, fde_offset);
177                if let Err(DwarfUnwinderError::UnwindInfoForAddressFailed(_)) = unwind_info {
178                    return Ok(UnwindResult::ExecRule(A::rule_if_uncovered_by_fde()));
179                }
180                let (unwind_info, encoding) = unwind_info?;
181                A::unwind_frame::<F, R, UCS, ES>(
182                    &debug_frame,
183                    unwind_info,
184                    encoding,
185                    regs,
186                    is_first_frame,
187                    read_stack,
188                )
189            }
190        }
191    }
192
193    fn unwind_info_for_fde<US: UnwindSection<R>>(
194        &mut self,
195        unwind_section: &US,
196        lookup_svma: u64,
197        fde_offset: u32,
198    ) -> Result<(&UnwindTableRow<R::Offset, UCS>, Encoding), DwarfUnwinderError> {
199        let fde = unwind_section.fde_from_offset(
200            &self.bases,
201            US::Offset::from(R::Offset::from_u32(fde_offset)),
202            US::cie_from_offset,
203        );
204        let fde = fde.map_err(DwarfUnwinderError::FdeFromOffsetFailed)?;
205        let encoding = fde.cie().encoding();
206        let unwind_info: &UnwindTableRow<_, _> = fde
207            .unwind_info_for_address(
208                unwind_section,
209                &self.bases,
210                self.unwind_context,
211                lookup_svma,
212            )
213            .map_err(DwarfUnwinderError::UnwindInfoForAddressFailed)?;
214        Ok((unwind_info, encoding))
215    }
216}

218pub(crate) fn base_addresses_for_sections<D>(
219    section_info: &mut impl ModuleSectionInfo<D>,
220) -> BaseAddresses {
221    let mut start_addr = |names: &[&[u8]]| -> u64 {
222        names
223            .iter()
224            .find_map(|name| section_info.section_svma_range(name))
225            .map(|r| r.start)
226            .unwrap_or_default()
227    };
228    BaseAddresses::default()
229        .set_eh_frame(start_addr(&[b"__eh_frame", b".eh_frame"]))
230        .set_eh_frame_hdr(start_addr(&[b"__eh_frame_hdr", b".eh_frame_hdr"]))
231        .set_text(start_addr(&[b"__text", b".text"]))
232        .set_got(start_addr(&[b"__got", b".got"]))
233}

252/// A binary search table for eh_frame FDEs. We generate this whenever a module
253/// without eh_frame_hdr is added.
254pub struct DwarfCfiIndex {
255    /// Contains the initial address for every FDE, relative to the base address.
256    /// This vector is sorted so that it can be used for binary search.
257    /// It has the same length as `fde_offsets`.
258    sorted_fde_pc_starts: Vec<u32>,
259    /// Contains the FDE offset for every FDE. The FDE at offset `fde_offsets[i]`
260    /// has a PC range which starts at `sorted_fde_pc_starts[i]`.
261    fde_offsets: Vec<u32>,
262}
263
264impl DwarfCfiIndex {
265    pub fn try_new<R, US>(
266        unwind_section: US,
267        bases: BaseAddresses,
268        base_svma: u64,
269    ) -> Result<Self, DwarfCfiIndexError>
270    where
271        R: Reader,
272        R::Offset: TryInto<u32>,
273        US: UnwindSection<R>,
274    {
275        let mut fde_pc_and_offset = Vec::new();
276
277        let mut cur_cie = None;
278        let mut entries_iter = unwind_section.entries(&bases);
279        while let Some(entry) = entries_iter.next()? {
280            let fde = match entry {
281                CieOrFde::Cie(cie) => {
282                    cur_cie = Some(cie);
283                    continue;
284                }
285                CieOrFde::Fde(partial_fde) => {
286                    partial_fde.parse(|unwind_section, bases, cie_offset| {
287                        if let Some(cie) = &cur_cie {
288                            if cie.offset()
289                                == <US::Offset as UnwindOffset<R::Offset>>::into(cie_offset)
290                            {
291                                return Ok(cie.clone());
292                            }
293                        }
294                        let cie = unwind_section.cie_from_offset(bases, cie_offset);
295                        if let Ok(cie) = &cie {
296                            cur_cie = Some(cie.clone());
297                        }
298                        cie
299                    })?
300                }
301            };
302            let pc = fde.initial_address();
303            let relative_pc = pc
304                .checked_sub(base_svma)
305                .ok_or(DwarfCfiIndexError::CouldNotSubtractBaseAddress)?;
306            let relative_pc = u32::try_from(relative_pc)
307                .map_err(|_| DwarfCfiIndexError::RelativeAddressTooBig)?;
308            let fde_offset = <R::Offset as TryInto<u32>>::try_into(fde.offset())
309                .map_err(|_| DwarfCfiIndexError::FdeOffsetTooBig)?;
310            fde_pc_and_offset.push((relative_pc, fde_offset));
311        }
312        fde_pc_and_offset.sort_by_key(|(pc, _)| *pc);
313        let sorted_fde_pc_starts = fde_pc_and_offset.iter().map(|(pc, _)| *pc).collect();
314        let fde_offsets = fde_pc_and_offset.into_iter().map(|(_, fde)| fde).collect();
315        Ok(Self {
316            sorted_fde_pc_starts,
317            fde_offsets,
318        })
319    }

321    pub fn try_new_eh_frame<D>(
322        eh_frame_data: &[u8],
323        section_info: &mut impl ModuleSectionInfo<D>,
324    ) -> Result<Self, DwarfCfiIndexError> {
325        let bases = base_addresses_for_sections(section_info);
326        let mut eh_frame = EhFrame::from(EndianSlice::new(eh_frame_data, LittleEndian));
327        eh_frame.set_address_size(8);
328
329        Self::try_new(eh_frame, bases, section_info.base_svma())
330    }

343    pub fn fde_offset_for_relative_address(&self, rel_lookup_address: u32) -> Option<u32> {
344        let i = match self.sorted_fde_pc_starts.binary_search(&rel_lookup_address) {
345            Err(0) => return None,
346            Ok(i) => i,
347            Err(i) => i - 1,
348        };
349        Some(self.fde_offsets[i])
350    }

357pub fn eval_cfa_rule<R: Reader, UR: DwarfUnwindRegs, S: EvaluationStorage<R>>(
358    section: &impl UnwindSection<R>,
359    rule: &CfaRule<R::Offset>,
360    encoding: Encoding,
361    regs: &UR,
362) -> Option<u64> {
363    match rule {
364        CfaRule::RegisterAndOffset { register, offset } => {
365            let val = regs.get(*register)?;
366            u64::try_from(i64::try_from(val).ok()?.checked_add(*offset)?).ok()
367        }
368        CfaRule::Expression(expr) => {
369            let expr = expr.get(section).ok()?;
370            eval_expr::<R, UR, S>(expr, encoding, regs)
371        }
372    }
373}
374
375fn eval_expr<R: Reader, UR: DwarfUnwindRegs, S: EvaluationStorage<R>>(
376    expr: Expression<R>,
377    encoding: Encoding,
378    regs: &UR,
379) -> Option<u64> {
380    let mut eval = Evaluation::<R, S>::new_in(expr.0, encoding);
381    let mut result = eval.evaluate().ok()?;
382    loop {
383        match result {
384            EvaluationResult::Complete => break,
385            EvaluationResult::RequiresRegister { register, .. } => {
386                let value = regs.get(register)?;
387                result = eval.resume_with_register(Value::Generic(value as _)).ok()?;
388            }
389            _ => return None,
390        }
391    }
392    let x = &eval.as_result().last()?.location;
393    if let Location::Address { address } = x {
394        Some(*address)
395    } else {
396        None
397    }
398}

400pub fn eval_register_rule<R, F, UR, S>(
401    section: &impl UnwindSection<R>,
402    rule: RegisterRule<R::Offset>,
403    cfa: u64,
404    encoding: Encoding,
405    val: u64,
406    regs: &UR,
407    read_stack: &mut F,
408) -> Option<u64>
409where
410    R: Reader,
411    F: FnMut(u64) -> Result<u64, ()>,
412    UR: DwarfUnwindRegs,
413    S: EvaluationStorage<R>,
414{
415    match rule {
416        RegisterRule::Undefined => None,
417        RegisterRule::SameValue => Some(val),
418        RegisterRule::Offset(offset) => {
419            let cfa_plus_offset =
420                u64::try_from(i64::try_from(cfa).ok()?.checked_add(offset)?).ok()?;
421            read_stack(cfa_plus_offset).ok()
422        }
423        RegisterRule::ValOffset(offset) => {
424            u64::try_from(i64::try_from(cfa).ok()?.checked_add(offset)?).ok()
425        }
426        RegisterRule::Register(register) => regs.get(register),
427        RegisterRule::Expression(expr) => {
428            let expr = expr.get(section).ok()?;
429            let val = eval_expr::<R, UR, S>(expr, encoding, regs)?;
430            read_stack(val).ok()
431        }
432        RegisterRule::ValExpression(expr) => {
433            let expr = expr.get(section).ok()?;
434            eval_expr::<R, UR, S>(expr, encoding, regs)
435        }
436        RegisterRule::Architectural => {
437            // Unimplemented
438            // TODO: Find out what the architectural rules for x86_64 and for aarch64 are, if any.
439            None
440        }
441        _ => None,
442    }
