1//! Simultaneously computes the sine and cosine of the number, `x`.
2
3use super::F32;
4
5impl F32 {
6    /// Simultaneously computes the sine and cosine of the number, `x`.
7    /// Returns `(sin(x), cos(x))`.
8    pub fn sin_cos(self) -> (Self, Self) {
9        (self.sin(), self.cos())
10    }
11}
12
